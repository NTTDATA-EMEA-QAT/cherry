$('[data-toggle="offcanvas"]').on('click', function (e) {

  e.preventDefault();

  $('body').toggleClass('offcanvas-show');
  
  if ($('body').hasClass('offcanvas-show')) {
    $('html').css('overflow', 'hidden');
  }
  else {
    $('html').css('overflow', 'visible');
  }
  
});
